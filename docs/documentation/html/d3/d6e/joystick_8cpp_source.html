164 joystick_position cJoystick::joystickPosition(int n)
165 {
166  joystick_position pos;
167 
168  if(n > -1 && n < m_axes)
169  {
170  int i0 = n * 2, i1 = n * 2 + 1;
171  float x0 = m_joystick_st->axis[i0] / 32767.0f,
172  y0 = -m_joystick_st->axis[i1] / 32767.0f;
173  float x = x0 * sqrt(1 - pow(y0, 2) / 2.0f),
174  y = y0 * sqrt(1 - pow(x0, 2) / 2.0f);
175 
176  pos.x = x0;
177  pos.y = y0;
178 
179  pos.theta = atan2(y, x);
180  pos.r = sqrt(pow(y, 2) + pow(x, 2));
181  }
182  else
183  {
184  pos.theta = pos.r = pos.x = pos.y = 0.0f;
185  }
186  return pos;
187 }
