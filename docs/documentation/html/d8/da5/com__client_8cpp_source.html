154  struct termios tty;

156  throw id + "[ERROR] Could not get the serial port settings.";
157 
158  tty.c_cflag &= ~PARENB; // Clear parity bit, disabling parity (most common)
159  tty.c_cflag &= ~CSTOPB; // Clear stop field, only 1 stop bit (most common)
160  tty.c_cflag &= ~CSIZE; // Clear all bits that set the data size
161  tty.c_cflag |= CS8; // 8 bits per byte (most common)
162  tty.c_cflag &= ~CRTSCTS; // Disable RTS/CTS hardware flow ctrl (most common)
163  tty.c_cflag |= CREAD | CLOCAL; // Turn on READ & ignore ctrlline(CLOCAL = 1)
164  tty.c_lflag &= ~ICANON;
165  tty.c_lflag &= ~ECHO; // Disable echo
166  tty.c_lflag &= ~ECHOE; // Disable erasure
167  tty.c_lflag &= ~ECHONL; // Disable new-line echo
168  tty.c_lflag &= ~ISIG; // Disable interpretation of INTR, QUIT and SUSP
169  tty.c_iflag &= ~(IXON | IXOFF | IXANY); // Turn off s/w flow ctrl
170  tty.c_iflag &= ~(IGNBRK | BRKINT | PARMRK | ISTRIP | INLCR | IGNCR |
171  ICRNL); // Disable any special handling of received bytes
172 
173  tty.c_oflag &= ~OPOST; // Prevent spe. interp. of out bytes (newline chars)
174  tty.c_oflag &= ~ONLCR; // Prevent conv of newline to car. ret/line feed
175  // tty.c_oflag &= ~OXTABS; // Prevent conversion of tabs to spaces (NOT PRESENT ON LINUX)
176  // tty.c_oflag &= ~ONOEOT; // Prevent removal of C-d chars (0x004) in output (NOT PRESENT ON LINUX)
177 
178  tty.c_cc[VTIME] = 4; // Wait for up to 1s, ret when any data is received.
179  tty.c_cc[VMIN] = 0;

181  // Set in/out baud rate to be 9600
