113 EMG::route_channel(uint8_t channel, uint8_t pos_in, uint8_t neg_in)
114 {
115  pos_in = (pos_in < 0) ? 0 : (pos_in > 6) ? 6 : pos_in;
116  neg_in = (neg_in < 0) ? 0 : (neg_in > 6) ? 6 : neg_in;
117 
118  channel = ((channel < 0) ? 0 : (channel > 2) ? 2 : channel);
119  uint8_t val = pos_in | (neg_in << 3);
120  m_regs[FLEX_CH0_CN_REG + channel] = val;
121  m_master->writeReg(m_id, FLEX_CH0_CN_REG + channel, val);
122 }

201 EMG::config_R2(uint8_t R2)
202 {
203  bool R3_val =
204  (m_regs[R3_RATE_CH0_REG] == 0b10 || m_regs[R3_RATE_CH0_REG] == 0b1000)
205  ? false
206  : true;
207  if(R2 < 5)
208  {
209  R2 = 0b0001; //4
210  m_fast_adc_max = 0x8000;
211  for(int i = 0; i < 3; i++)

213  }
214  else if(R2 < 6)

216  R2 = 0b0010; //5
217  m_fast_adc_max = 0xC350;
218  for(int i = 0; i < 3; i++)

220  }
221  else if(R2 < 8)

223  R2 = 0b0100; //6
224  m_fast_adc_max = 0xF300;

227  }
228  else
229  {
230  R2 = 0b1000; //8
231  m_fast_adc_max = 0x8000;
232  for(int i = 0; i < 3; i++)

241 {

265  else
266  m_precise_adc_max[ch] = R3_val ? 0x800000 : 0xF30000;
267 
268  m_regs[R3_RATE_CH0_REG + ch] = R3;
269  m_master->writeReg(m_id, R3_RATE_CH0_REG + ch, R3);
